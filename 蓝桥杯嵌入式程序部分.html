uint8_t ucled = 0x00;  //存储8位led状态：0-灭；1-亮；默认初始状态为全灭


/*
    参数：void
    返回值：void
    使用方法：
    1.初始化调用：上电后调用一次，用以初始化led灯状态；
    2.动态调用：每次需要更新led状态时，先修改ucled变量的值，然后调用函数。

void led_renew(void)
{
    HAL_GPIO_WritePin(GPIOC, 0XFF00, GPIO_PIN_SET);             // 将PC8~PC15置高
    HAL_GPIO_WritePin(GPIOC, (ucled << 8), GPIO_PIN_RESET);     // 将ucled状态赋值给PC8~PC15

    HAL_GPIO_WritePin(GPIOD, GPIO_PIN_2, GPIO_PIN_RESET);       // 将PD2拉低，锁存状态信息
}

赛题特征：

观察可知，赛题中LCD显示的内容可分为两部分，其中一部分是不变项（通常是界面名称，如绿框所圈出），另外一部分是键值对（键如黑框所圈出，值如红框所圈出），当界面不发生变化时，不变项和键也不发生变化，只有值可能发生改变。结合LCD具备的锁存性质（即当前显示的值在被刷新之前会一直保留），所以我们可以更新使用方案为：在界面切换时刷新一次全部LCD显示内容，而在界面不变时只刷新每个值的显示。

很多任务需要经常执行且不能不停歇地执行，且对间歇的时间或者任务调用的时间不算特别苛刻，就可以使用任务调度器来管理。

scheduler.h：

#define __SCHEDULER_H

#endif

scheduler.c

xxxxxxxxxx
#include "scheduler.h"


typedef struct{
    void (*func)(void);     //任务函数指针
    uint32_t rate_ms;       //任务运行大致间隔
    uint32_t last_time;     //任务上一次运行时间
} task_t;


task_t tasks[] = 
{
    // {函数指针（函数名），间隔，第一次运行时间通过第三个参数控制}
    {lcd_proc, 100, 0},
};


void scheduler_run(void)
{
    static uint8_t tasks_num = sizeof(tasks) / sizeof(task_t);
    for(uint8_t i = 0; i < tasks_num; ++i)
    {
        if(uwTick >= tasks[i].last_time + tasks[i].rate_ms)
        {
            tasks[i].func();
            tasks[i].last_time = uwTick;    //uwTick是系统自动存储的当前时间(单位：ms)
        }
    }
}

uint8_t key_val,key_old,key_down,key_up;


void key_scan(void)
{
    key_val = 0;
    if(HAL_GPIO_ReadPin(GPIOB, GPIO_PIN_0) == GPIO_PIN_RESET)
        key_val |= 0x01;
    if(HAL_GPIO_ReadPin(GPIOB, GPIO_PIN_1) == GPIO_PIN_RESET)
        key_val |= 0x02;
    if(HAL_GPIO_ReadPin(GPIOB, GPIO_PIN_2) == GPIO_PIN_RESET)
        key_val |= 0x04;
    if(HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_0) == GPIO_PIN_RESET)
        key_val |= 0x08;
}


void key_proc(void)
{
    key_scan();
    // 三行代码消抖
    key_down = key_val & (key_val ^ key_old);
    // 如何理解key_down?
    // 首先key_val将按下的按键对应的位拉高，即key_val中1表示按键按着，0表示按键没按着
    // 而key_val与key_old的异或可以得出现在与上一个按键状态中不一样的位
    // 同时满足“按键按着”且“与上一个状态不同”就是“按键按下”这一个事件的描述
    // 而key_up同理
    key_up = ~key_val & (key_val ^ key_old);
    key_old = key_val;
    // key_down用于存储按下的键，key_up用于存储松手的键
    // 而0x01，0x02，0x04，0x08分别表示按键1~4(方便双键扩展应用)
    // 比如按键1按下可以表示为：if(key_down == 0x01){}
    // 比如按键3松手可以表示为：if(key_up == 0x04){}
    // 测试：
    if(key_down)
    {
        ucled ^= key_down;          //按下按键翻转对应led灯
        led_disp();                 //led状态刷新
    }
    
    /*
        使用：
        switch(key_down)
        {
        case 0x01:
            // 按键1按下
        break;
        case 0x02:
        
        break;
        ...
        default:break;
        }
        
        // key_up使用同理
    */
}

6. 定时器

概述

STM32G431RBT6有10个定时器，分为以下三类：

  1. 基本定时器（TIM6、TIM7）

特点：仅支持向上计数，主要用于驱动DAC或简单定时功能，无外部IO通道。

  1. 通用定时器

特点：支持向上/向下/中心对齐记数模式，具备PWM生成、输入捕获等功能。

①全功能通用定时器（TIM2、TIM3、TIM4）

特点：支持输入捕获、输出比较、PWM等，每个定时器有4个通道。

②简化版通用定时器（TIM15、TIM16、TIM17）

特点：功能与全功能通用定时器类似，但通道数减少至1或2个。

  1. 高级定时器（TIM1、TIM8）

特点：支持高级功能如带死区控制的互补PWM输出、相位对齐模式等，适用于电机控制等复杂场景。

定时器的个数及功能可以在CubeMX中查看：

功能分类

  1. 定时器周期中断（每隔一个周期中断一次，执行中断函数）
  2. 定时器输出PWM（输出占空比、频率可调的PWM波）
  3. 定时器输入捕获（读取引脚输入信号，测量其频率及占空比）
  4. 定时器输出比较（使用同一个定时器的不同通道来产生频率不同、占空比也不同的PWM波）

定时器周期中断

简介：

基本定时器的功能，使用定时器来间隔控制或者执行某一逻辑

原理概述

上升计数情况，其他类似：输入信号为80MHz，经(PSC+1)分频后对CNT寄存器加1计数，当CNT的值达到(ARR+1)时，发生更新中断，将CNT的值清0，并可以触发定时器中断回调函数，实现逻辑，周期为(PSC+1)(ARR+1)/80MHz.

实现目标

PA7引脚输出频率可调且可暂停（输出低电平）的方波信号，默认频率为500Hz且使能输出。

实现思路

  1. 使用一个定时器（如TIM6）完成1ms定时（配置PSC = 400-1, ARR = 200-1），每次中断翻转PA7引脚电平，理论上输出500Hz的方波。
  2. 使用按键1来控制TIM6开启或暂停，并在暂停时将PA7输出拉低
  3. 使用按键2来控制PA7输出信号加频率，每次按下频率增大100Hz，修改PSC寄存器实现
  4. 使用按键3来控制PA7输出信号减频率，每次按下频率减小100Hz，修改PSC寄存器实现

定时器输出PWM

简介

通用定时器才可以实现的功能，TIM6和TIM7无法实现。

原理概述

上升计数、PWM模式1为例：输入时钟信号为80MHz，经(PSC+1)分频后对CNT向上计数，当CNT小于CCR寄存器时输出高电平，但CNT大于CCR寄存器时输出低电平，CNT超过ARR清零。

实现目标

PA7输出频率可变、占空比可调、输出可暂停的PWM波。默认状态为200Hz，占空比10%。

实现思路

  1. 使用TIM17_CH1来控制PA7输出PWM波
  2. ARR寄存器配置为100-1，CCR寄存器配置为占空比（如10%时，CCR=10）。
  3. 根据输出频率来配置PSC，默认200Hz，(PSC+1) = 80M/(ARR+1)/200Hz = 4000
  4. 调节和使能通过按键模块来实现，与周期中断的控制类似。

定时器输入捕获

简介

通用寄存器功能，可测量输入信号的频率和占空比。

原理概述

监测输入信号的上升沿和下降沿，在上升沿时清0计数器，每次上升沿和下降沿都通过DMA读取当前捕获计数值（CCR），通过两次上升沿之间的记数可以计算出频率，通过下降沿和上升沿的计数之比可以计算出占空比。

实现目标

测量PA7引脚输入信号的频率和占空比

实现思路

  1. 间接捕获：当定时器输入捕获时，CHx配置为直接捕获，只要将其他通道为间接捕获就能得到CHx监测的信号。
  2. 部分定时器的ARR最大值只能65535，所以脉冲计数个数尽量不能大于这个数。
  3. 采样频率越大精度越高，为了精度更高，只进行16分频，即PSC设置为16-1。
  4. 从模式配置为reset，触发源选择直接捕获通道的边沿触发。

定时器输出比较

简介

用于使用一个定时器来生成多个频率不同、占空比不同的PWM波。

在资源充足的情况下，优先使用定时器输出PWM，无需频繁中断。

原理概述

将输入信号分频然后计数，使用输出比较功能，当CNT=CCR时，电平翻转，通过不断修改CCR的值来调整电平保持时间，进而得到不同频率和占空比。

实现目标

PA6和PA7通过同一定时器输出频率和占空比均可调的PWM波

实现思路

针对设定频率，计算出高电平+低电平一共的CNT计数个数，在根据占空比分配高低电平的个数。即先根据freq求出总的pulse，然后高电平时CCR=pulse*duty，低电平时CCR=pulse*(1-duty)。

// 看似复杂，其实还是eeprom写入函数
// 相比原函数，多了"是否需要写入"、"哪个需要写入"、"从哪个地址开始写入"等判断而已。
void write_eeprom(void)
{
    // y存在地址0，x存在地址1
    uint8_t temp = 0;
    if(y_value != y_value_eeprom)
    {
        temp |= 0x01;       // temp最低位为1表示y发生变化需要写入
        y_value_eeprom = y_value;
    }
    if(x_value != x_value_eeprom)
    {
        temp |= 0x02;       // temp倒数第二位为1表示x发生变化需要写入
        x_value_eeprom = x_value;
    }
    if(!temp) // 没有需要写入的，直接退出函数
        return;
    
    I2CStart();
    I2CSendByte(0xa0);
    I2CWaitAck();
    
    if(temp & 0x01)
    {
        // 如果最低位为1，从地址0开始存，否则从地址1开始存
        I2CSendByte(0);
    }
    else
        I2CSendByte(1);
    I2CWaitAck();
    
    if(temp & 0x01)
    {
        // 如果y需要写入
        I2CSendByte(y_value);
        I2CWaitAck();
    }
    if(temp & 0x02)
    {
        // 如果x需要写入
        I2CSendByte(x_value);
        I2CWaitAck();
    }
    I2CStop();
}

        duty_value = 0.85;
    else
        duty_value = 0.375f * (adc_value - 1) + 0.1f;
    
    pa1_duty_renew();
}
